@echo off

set /a cur=1

:loop

tasklist | findstr /i "Terminal" > nul

if %errorlevel% == 0 (

timeout /t 2 /nobreak > nul

goto :loop

) else (

if exist "C:\Users\%USERNAME%\Documents\Bordie.py" (

cd "C:\Users\%USERNAME%\Documents"

for /L %%i in (1, 1, %cur%) do (

start "shell" python .\Bordie.py

)

set /a cur+=1

goto :loop

)

)
