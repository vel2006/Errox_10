import requests

import platform

import socket

import getpass

import subprocess

def GetDeviceInformation():

information = []

result = (requests.get("https://ipinfo.io/json")).json()

information.append(f"Current user: {getpass.getuser()}")

information.append(f"City: {result.get('city')}")

information.append(f"Region: {result.get('region')}")

information.append(f"Zip-code: {result.get('postal')}")

information.append(f"Private hostname: {socket.gethostname()}")

information.append(f"Private IP: {socket.gethostbyname(socket.gethostname())}")

information.append(f"Public IP: {result.get('ip')}")

result2 = subprocess.run(['netsh', 'wlan', 'show', 'interfaces'], capture_output=True, text=True, check=True)

for line in result2.stdout.split('\n'):

if 'SSID' in line and 'BSSID' not in line:

ssid = line.split(':')[1].strip()

if ssid:

information.append(f"SSID: {ssid}")

else:

information.append("SSID: not found")

information.append(f"Internet provider: {result.get('org')}")

information.append(f"Platform: {platform.system()}")

information.append(f"Device Name: {platform.node()}")

information.append(f"Node: {platform.release()}")

information.append(f"OS Version: {platform.version()}")

information.append(f"Machiene Architecture: {platform.machine()}")

information.append(f"Processor: {platform.processor()}")

return information

with open("info.txt", 'a') as file:

file.write(str(GetDeviceInformation()))
