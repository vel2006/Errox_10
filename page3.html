import requests

import os

def GetInbetween(input, firstOccurring, secondOccurring):

startIndex = input.find(firstOccurring)

endIndex = input.find(secondOccurring)

if startIndex != -1 and endIndex != -1:

return input[startIndex + len(startIndex):endIndex]

def DownloadFromWeb(link, outputFile):

response = requests.get(link)

if response.status_code != 200:

print(f"Cannot access {link} for creating {outputFile}")

return 1

source = GetInbetween(response.text, "", "")

for line in source.splitlines():

if "

" in line:

path = os.path.join(os.getcwd(), outputFile)

with open(path, 'a') as file:

file.write(GetInbetween(line, "

", "

"))

return 0
